# declares
int x;
string y;
char c;
int z;

# Assignments
x = 42;
y = "Hello, Parser!";
c = 'A';
z = x + 10 * 3;


# Print statement function call
print(x);
print("A String Literal");
print('A');

# Control flow
if (x > 20) {
    print("X is large!");
} else {
    print("X is small!");
}

# While loop
while (x > 0) {
    x = x - 1;
}

# Repeat-Until loop
repeat {
    x = x + 1;
} until (x == 10);

# Factorial function call
z = $(5);
z = $(1+2+3);
