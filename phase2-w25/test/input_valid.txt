# declares
int a;
string b;
char c;

# Assignments
a = 42;
b = "Hello, Parser!";
c = 'A';
a = a + 10 * 3;

# Print statement function call
print(a);
print("A String Literal");
print('A');

# Control flow
if (a > 20) {
    print("large!");
} else {
    print("small!");
}

# While loop
while (a > 0) {
    a = a - 1;
}

# Repeat-Until loop
repeat {
    int d;
    d = a + 1;
    a = a + d;
} until (a >= 10);

# Factorial function call
int z;
z = $(5);
